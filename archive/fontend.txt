        <p id="errorMessage" class="error-message">密码错误，请重试</p>
        <p class="help-text">提示：密码为四位数字，请向项目创建者索取</p>

    <script>
      document.addEventListener("DOMContentLoaded", function () {
        // 密码输入处理
        const digitInputs = document.querySelectorAll(".digit-input");
        const submitButton = document.getElementById("passwordButton");
        const errorMessage = document.getElementById("errorMessage");
        const projectId = "69758d5b53ace4a187c60b6e";
        const domain = "html2web.com";
        const fulldomain = "www.html2web.com";
        const allowDownload = "false";
        // 自动聚焦到下一个输入框
        digitInputs.forEach((input, index) => {
          input.addEventListener("input", function () {
            // 检查是否是数字
            if (!/^\d*$/.test(this.value)) {
              this.value = "";
              return;
            }

            // 填充后自动切换到下一个输入框
            if (this.value && index < digitInputs.length - 1) {
              digitInputs[index + 1].focus();
            }

            // 检查是否所有位置都已填写
            checkAllDigits();
          });

          // 处理删除键
          input.addEventListener("keydown", function (e) {
            if (e.key === "Backspace" && !this.value && index > 0) {
              digitInputs[index - 1].focus();
            }
          });
        });

        // 检查所有数字是否已输入
        function checkAllDigits() {
          let isComplete = true;
          let password = "";

          digitInputs.forEach((input) => {
            if (!input.value) {
              isComplete = false;
            }
            password += input.value;
          });

          // 启用/禁用提交按钮
          submitButton.disabled = !isComplete;

          return password;
        }

        // 提交密码
        submitButton.addEventListener("click", function () {
          const password = checkAllDigits();
          submitButton.disabled = true;

          // 调用后端API验证密码
          fetch(`//${fulldomain}/api/verify-password/${projectId}`, {
            method: "POST",
            headers: {
              "Content-Type": "application/json",
            },
            body: JSON.stringify({ password }),
          })
            .then((response) => response.json())
            .then((data) => {
              if (data.success) {
                // 密码正确，加载iframe内容
                errorMessage.classList.remove("show");

                // 将验证成功的密码保存到本地存储，避免用户刷新页面后再次输入
                try {
                  localStorage.setItem(
                    `project_password_${projectId}`,
                    password
                  );
                } catch (e) {
                  console.error("无法保存密码到本地存储", e);
                }
                var html =
                  '<iframe id="projectFrame" src="//' +
                  fulldomain +
                  "/view/" +
                  projectId +
                  "?password=" +
                  password +
                  '" allow="microphone; camera" frameborder="0" width="100%" height="100%"></iframe>';
                document.body.innerHTML = html;
              } else {
                // 密码错误，显示错误信息
                errorMessage.classList.add("show");

                // 清空所有输入框并聚焦到第一个
                digitInputs.forEach((input) => {
                  input.value = "";
                });
                digitInputs[0].focus();
                submitButton.disabled = false;
              }
            })
            .catch((error) => {
              console.error("验证密码时出错:", error);
              errorMessage.textContent = "验证过程中发生错误，请稍后重试";
              errorMessage.classList.add("show");
              submitButton.disabled = false;
            });
        });

        // 检查本地存储中是否有之前验证成功的密码
        try {
          const savedPassword = localStorage.getItem(
            `project_password_${projectId}`
          );
          if (savedPassword) {
            // 自动验证已保存的密码
            fetch(`//${fulldomain}/api/verify-password/${projectId}`, {
              method: "POST",
              headers: {
                "Content-Type": "application/json",
              },
              body: JSON.stringify({ password: savedPassword }),
            })
              .then((response) => response.json())
              .then((data) => {
                if (data.success) {
                  // 密码正确，加载iframe内容
                  var html =
                    '<iframe id="projectFrame" src="//' +
                    fulldomain +
                    "/view/" +
                    projectId +
                    "?password=" +
                    savedPassword +
                    '" allow="microphone; camera" frameborder="0" width="100%" height="100%"></iframe>';
                  document.body.innerHTML = html;
                }
              })
              .catch((error) => {
                console.error("验证已保存密码时出错:", error);
                // 验证失败时不显示错误，等待用户手动输入
              });
          }
        } catch (e) {
          console.error("无法访问本地存储", e);
        }
      });
    </script>
